library(corrplot)
library(Hmisc)

my.cors <- as.matrix(small) #creating matrix with variables of interest
mcor <- cor(my.cors, use = "pairwise.complete.obs")

col <- colorRampPalette(c("#BB4444", "#EE9988", "#FFFFFF", "#77AADD", "#4477AA"))
#picking color palette: http://www.color-hex.com/

my.cors.p.sig <- Hmisc::rcorr(my.cors, type="pearson")$P

corrplot(mcor, method="shade", shade.col=NA, tl.col="black", tl.srt=2, tl.cex =1, 
         cl.cex = 2, number.cex= 1,
         col=col(200), addCoef.col="black", addcolorlabel="no", 
         order="AOE",p.mat=my.cors.p.sig,sig.level=.05, insig="blank",type="lower")
